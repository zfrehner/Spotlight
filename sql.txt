// mysql code for skateboard

// creating the data tables
// gym 1
    CREATE TABLE gym1 (
    	id int(11) not null AUTO_INCREMENT PRIMARY KEY,
        first varchar(20) not null,
        last varchar(20) not null,
        gender char(1),
        age int(3) not null,
        city varchar(20),
        state varchar(20),
        comment text,
        time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    );
